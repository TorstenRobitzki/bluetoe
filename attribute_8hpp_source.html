146  /*
147  * An attribute is an uuid combined with a mean of how to access the attributes
148  *
149  * design decisions to _not_ use pointer to staticaly allocated virtual objects:
150  * - makeing access a pointer to a virtual base class would result in storing a pointer that points to a pointer (vtable) to list with a bunch of functions.
151  * - it's most likely that most attributes will not have any mutable data.
152  * attribute contains only one function that takes a attribute_access_type to save memory and in the expectation that there are only a few different combination of
153  * access functions.
154  */
155  struct attribute {
156  // all uuids used by GATT are 16 bit UUIDs (except for Characteristic Value Declaration for which the value internal_128bit_uuid is used, if the UUID is 128bit long)
157  std::uint16_t uuid;
158  attribute_access access;
159  };

367  inline attribute_access_result attribute_value_read_access( attribute_access_arguments& args, const std::uint8_t* memory, std::size_t size )
368  {
369  if ( args.type == attribute_access_type::compare_value )
370  {
371  return size == args.buffer_size && std::equal( memory, memory + size, args.buffer )
372  ? attribute_access_result::value_equal
373  : attribute_access_result::read_not_permitted;
374  }
375 
376  if ( args.buffer_offset > size )
377  return details::attribute_access_result::invalid_offset;
378 
379  args.buffer_size = std::min< std::size_t >( args.buffer_size, size - args.buffer_offset );
380  const std::uint8_t* const ptr = memory + args.buffer_offset;
381 
382  std::copy( ptr, ptr + args.buffer_size, args.buffer );
383 
384  return details::attribute_access_result::success;
385  }

387  inline attribute_access_result attribute_value_read_only_access( attribute_access_arguments& args, const std::uint8_t* memory, std::size_t size )
388  {
389  if ( args.type != attribute_access_type::read && args.type != attribute_access_type::compare_value )
390  return attribute_access_result::write_not_permitted;
391 
392  return attribute_value_read_access( args, memory, size );
