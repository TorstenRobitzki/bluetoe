191  template < class BufferedRadio, std::size_t MTUSize >
192  write_buffer ll_l2cap_sdu_buffer< BufferedRadio, MTUSize >::next_ll_l2cap_received()
193  {
194  // is there already a defragmented L2CAP SDU?
195  if ( receive_buffer_used_ != 0 && receive_size_ == 0 )
196  return { receive_buffer_, receive_buffer_used_ };
197 
198  for ( auto pdu = this->next_received(); pdu.size; pdu = this->next_received() )
199  {
200  const std::uint16_t header = layout::header( pdu );
201  const std::uint16_t type = header & pdu_type_mask;
202 
203  // link layer control PDUs are not fragmented by default
204  if ( type == pdu_type_link_layer )
205  return pdu;
206 
207  // l2cap message
208  const auto body = layout::body( pdu );
209  const std::size_t body_size = body.second - body.first;
210 
211  if ( type == pdu_type_start )
212  {
213  if ( body_size >= l2cap_header_size )
214  {
215  const std::uint16_t l2cap_size = bluetoe::details::read_16bit( body.first );
216 
217  // short l2cap PDU that is not fragmented
218  if ( l2cap_size + l2cap_header_size == body_size )
219  return pdu;
220 
221  if ( l2cap_size <= MTUSize )
222  {
223  receive_size_ = l2cap_size + overall_overhead;
224  add_to_receive_buffer( pdu.buffer, pdu.buffer + pdu.size );
225  }
226  }
227  }
228  else
229  {
230  add_to_receive_buffer( body.first, body.second );
231  }
232 
233  // consume LL PDU
234  this->free_received();
235 
236  // is there now a defragmented L2CAP SDU?
237  if ( receive_buffer_used_ != 0 && receive_size_ == 0 )
238  return { receive_buffer_, receive_buffer_used_ };
239  }
240 
241  return { nullptr, 0 };
242  }

254  template < class BufferedRadio, std::size_t MTUSize >
255  void ll_l2cap_sdu_buffer< BufferedRadio, MTUSize >::try_send_pdus()
256  {
257  while ( transmit_size_ )
258  {
259  const bool first_fragment = transmit_buffer_used_ == 0;
260 
261  // for the first PDU, the header overhead is already taken into account. For all additonal fragments,
262  // an additional header has to be allocated.
263  const std::size_t overhead = first_fragment ? 0 : ll_overhead;
264  const auto buffer = this->allocate_transmit_buffer( std::min( transmit_size_ + overhead, this->max_tx_size() ) );
265 
266  if ( buffer.size == 0 )
267  return;
268 
269  if ( first_fragment )
270  {
271  // The first fragment contains the original LL header
272  const auto copy_size = std::min< std::size_t >( buffer.size, transmit_size_ );
273 
274  std::copy( &transmit_buffer_[ 0 ], &transmit_buffer_[ copy_size ], buffer.buffer );
275  layout::header( buffer, pdu_type_start | ( ( copy_size - ll_overhead ) << 8 ) );
276 
277  transmit_size_ -= copy_size;
278  transmit_buffer_used_ += copy_size;
279  }
280  else
281  {
282  // for every additional fragment, an additional header has to be generated
283  const auto body = layout::body( buffer );
284  const auto copy_size = std::min< std::size_t >( std::distance( body.first, body.second ), transmit_size_ );
285 
286  std::copy( &transmit_buffer_[ transmit_buffer_used_ ], &transmit_buffer_[ transmit_buffer_used_+ copy_size ], body.first );
287  layout::header( buffer, pdu_type_continuation | ( copy_size << 8 ) );
288 
289  transmit_size_ -= copy_size;
290  transmit_buffer_used_ += copy_size;
291  }
292 
293  this->commit_transmit_buffer( buffer );
294  }
295 
296  transmit_buffer_used_ = 0;
297  }
