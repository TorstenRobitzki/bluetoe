397 template < typename ... Options >
398 template < typename ConnectionData >
399 void server< Options... >::l2cap_input( const std::uint8_t* input, std::size_t in_size, std::uint8_t* output, std::size_t& out_size, ConnectionData& connection )
400 {
401 // clip the output size to the negotiated mtu
402 out_size = std::min< std::size_t >( out_size, connection.negotiated_mtu() );
403
404 assert( in_size != 0 );
405 assert( out_size >= details::default_att_mtu_size );
406
407 const details::att_opcodes opcode = static_cast< details::att_opcodes >( input[ 0 ] );
408
409 switch ( opcode )
410 {
411 // do not respond to an error response:
412 case details::att_opcodes::error_response:
413 out_size = 0;
414 break;
415
416 case details::att_opcodes::exchange_mtu_request:
417 handle_exchange_mtu_request( input, in_size, output, out_size, connection );
418 break;
419 case details::att_opcodes::find_information_request:
420 handle_find_information_request( input, in_size, output, out_size );
421 break;
422 case details::att_opcodes::find_by_type_value_request:
423 handle_find_by_type_value_request( input, in_size, output, out_size );
424 break;
425 case details::att_opcodes::read_by_type_request:
426 handle_read_by_type_request( input, in_size, output, out_size, connection );
427 break;
428 case details::att_opcodes::read_request:
429 handle_read_request( input, in_size, output, out_size, connection );
430 break;
431 case details::att_opcodes::read_blob_request:
432 handle_read_blob_request( input, in_size, output, out_size, connection );
433 break;
434 case details::att_opcodes::read_by_group_type_request:
435 handle_read_by_group_type_request( input, in_size, output, out_size );
436 break;
437 case details::att_opcodes::read_multiple_request:
438 handle_read_multiple_request( input, in_size, output, out_size, connection );
439 break;
440 case details::att_opcodes::write_request:
441 handle_write_request( input, in_size, output, out_size, connection );
442 break;
443 case details::att_opcodes::write_command:
444 handle_write_command( input, in_size, output, out_size, connection );
445 break;
446 case details::att_opcodes::prepare_write_request:
447 handle_prepair_write_request( input, in_size, output, out_size, connection, write_queue_type() );
448 break;
449 case details::att_opcodes::execute_write_request:
450 handle_execute_write_request( input, in_size, output, out_size, connection, write_queue_type() );
451 break;
452 case details::att_opcodes::confirmation:
453 handle_value_confirmation( input, in_size, output, out_size, connection );
454 break;
455 default:
456 error_response( *input, details::att_error_codes::request_not_supported, output, out_size );
457 break;
458 }

463 template < bool >
464 struct copy_name
465 {
466 static std::uint8_t* impl( std::uint8_t* begin, std::uint8_t* end, const char* const name )
467 {
468 if ( ( end - begin ) <= 2 )
469 return begin;
470
471 const std::size_t name_length = std::strlen( name );
472 const std::size_t max_name_len = std::min< std::size_t >( name_length, end - begin - 2 );
473
474 if ( name_length > 0 )
475 {
476 begin[ 0 ] = max_name_len + 1;
477 begin[ 1 ] = max_name_len == name_length
478 ? bits( details::gap_types::complete_local_name )
479 : bits( details::gap_types::shortened_local_name );
480
481 std::copy( name + 0, name + max_name_len, &begin[ 2 ] );
482 begin += max_name_len + 2;
483 }
484
485 return begin;
486 }
487 };

494 return begin;

707 template < typename ... Options >
708 void server< Options... >::notification_output( std::uint8_t* output, std::size_t& out_size, connection_data& connection, const details::notification_data& data )
709 {
710 assert( data.valid() );
711
712 if ( connection.client_configurations().flags( data.client_characteristic_configuration_index() ) & details::client_characteristic_configuration_notification_enabled &&
713 out_size >= 3 )
714 {
715 auto read = details::attribute_access_arguments::read( output + 3, output + out_size, 0, connection.client_configurations(), connection.security_attributes(), this );
716 auto attr = attribute_at( data.attribute_table_index() );
717 auto rc = attr.access( read, data.attribute_table_index() );
718
719 if ( rc == details::attribute_access_result::success )
720 {
721 *output = bits( details::att_opcodes::notification );
722 details::write_handle( output +1, handle_mapping::handle_by_index( data.attribute_table_index() ) );
723
724 out_size = 3 + read.buffer_size;
725 }
726 else
727 {
728 out_size = 0;
729 }
730 }
731 else
732 {
733 out_size = 0;
734 }
735 }

906 template < typename ... Options >
907 void server< Options... >::handle_find_information_request( const std::uint8_t* input, std::size_t in_size, std::uint8_t* output, std::size_t& out_size )
908 {

910
911 if ( !check_size_and_handle_range< 5u >( input, in_size, output, out_size, starting_handle, ending_handle ) )
912 return;

916
917 std::size_t ending_index = handle_mapping::first_index_by_handle( ending_handle );
918
919 // if the ending handle points not on an existing attribute, the search will end at the next, lower handle
920 if ( ending_index != details::invalid_attribute_index && handle_mapping::handle_by_index( ending_index ) != ending_handle )
921 --ending_index;

927 ++write_ptr;

929 if ( write_ptr != write_end )

932 only_16_bit_uuids
933 ? details::att_uuid_format::short_16bit
934 : details::att_uuid_format::long_128bit );
935
936 ++write_ptr;
937
938 }
939
940 write_ptr = collect_handle_uuid_tuples( start_index, ending_index, only_16_bit_uuids, write_ptr, write_end );
941
942 out_size = write_ptr - &output[ 0 ];
943 }
