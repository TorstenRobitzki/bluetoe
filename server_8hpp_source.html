283  void server< Options... >::l2cap_input( const std::uint8_t* input, std::size_t in_size, std::uint8_t* output, std::size_t& out_size, connection_data& connection )
284  {
285  // clip the output size to the negotiated mtu
286  out_size = std::min< std::size_t >( out_size, connection.negotiated_mtu() );
287 
288  assert( in_size != 0 );
289  assert( out_size >= details::default_att_mtu_size );
290 
291  const details::att_opcodes opcode = static_cast< details::att_opcodes >( input[ 0 ] );
292 
293  switch ( opcode )
294  {
295  case details::att_opcodes::exchange_mtu_request:
296  handle_exchange_mtu_request( input, in_size, output, out_size, connection );
297  break;
298  case details::att_opcodes::find_information_request:
299  handle_find_information_request( input, in_size, output, out_size, connection );
300  break;
301  case details::att_opcodes::find_by_type_value_request:
302  handle_find_by_type_value_request( input, in_size, output, out_size, connection );
303  break;
304  case details::att_opcodes::read_by_type_request:
305  handle_read_by_type_request( input, in_size, output, out_size, connection );
306  break;
307  case details::att_opcodes::read_request:
308  handle_read_request( input, in_size, output, out_size, connection );
309  break;
310  case details::att_opcodes::read_blob_request:
311  handle_read_blob_request( input, in_size, output, out_size, connection );
312  break;
313  case details::att_opcodes::read_by_group_type_request:
314  handle_read_by_group_type_request( input, in_size, output, out_size, connection );
315  break;
316  case details::att_opcodes::read_multiple_request:
317  handle_read_multiple_request( input, in_size, output, out_size, connection );
318  break;
319  case details::att_opcodes::write_request:
320  handle_write_request( input, in_size, output, out_size, connection );
321  break;
322  case details::att_opcodes::write_command:
323  handle_write_command( input, in_size, output, out_size, connection );
324  break;
325  case details::att_opcodes::prepare_write_request:
326  handle_prepair_write_request( input, in_size, output, out_size, connection, write_queue_type() );
327  break;
328  case details::att_opcodes::execute_write_request:
329  handle_execute_write_request( input, in_size, output, out_size, connection, write_queue_type() );
330  break;
331  default:
332  error_response( *input, details::att_error_codes::request_not_supported, output, out_size );
333  break;
334  }
