45 static void start_high_frequency_clock()
46 {
47 // This tasks starts the high frequency crystal oscillator (HFXO)
48 nrf_clock->TASKS_HFCLKSTART = 1;
49
50 // TODO: do not wait busy
51 // Issue: do not poll for readiness of the high frequency clock #63
52 while ( !nrf_clock->EVENTS_HFCLKSTARTED )
53 ;
54
55 nrf_clock->EVENTS_HFCLKSTARTED = 0;
56 }

58 inline void start_lfclock_and_rtc()
59 {
60 nrf_clock->EVENTS_LFCLKSTARTED = 0;
61 nrf_clock->TASKS_LFCLKSTART = 1;
62
63 while ( nrf_clock->EVENTS_LFCLKSTARTED == 0 )
64 ;
65
66 // https://infocenter.nordicsemi.com/topic/errata_nRF52840_Rev3/ERR/nRF52840/Rev3/latest/anomaly_840_20.html#anomaly_840_20
67 nrf_rtc->TASKS_STOP = 0;
68 nrf_rtc->TASKS_START = 1;
69
70 // Configure the RTC to generate these two events
71 // Overflow flag does not harm the power performance much and is used for
72 // debugging.
73 nrf_rtc->EVTEN =
74 ( RTC_EVTEN_COMPARE0_Enabled << RTC_EVTEN_COMPARE0_Pos )
75 | ( RTC_EVTEN_COMPARE1_Enabled << RTC_EVTEN_COMPARE1_Pos )
76 | ( RTC_EVTEN_OVRFLW_Enabled << RTC_EVTEN_OVRFLW_Pos );
77 }
