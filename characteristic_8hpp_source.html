248  static void fixup_auto_uuid( details::attribute_access_arguments& args )
249  {
250  // needs a 16 bit characteristic index!!!
251  std::uint16_t char_index = 1;
252 
253  static constexpr std::size_t uuid_offset = 3;
254 
255  int index_low = static_cast< int >( args.buffer_offset ) - static_cast< int >( uuid_offset );
256  int index_high = static_cast< int >( args.buffer_offset ) - static_cast< int >( uuid_offset + 1 );
257 
258  if ( index_low >= 0 && index_low < static_cast< int >( args.buffer_size ) )
259  args.buffer[ index_low ] ^= ( char_index & 0xff );
260 
261  if ( index_high >= 0 && index_high < static_cast< int >( args.buffer_size ) )
262  args.buffer[ index_high ] ^= ( char_index >> 8 );
263  }
264 
265  /*
266  * the characteristic decalarion consists of 3 parts a Properties byte, two bytes index and 2 - 16 bytes UUID
267  */
268  static details::attribute_access_result char_declaration_access( details::attribute_access_arguments& args, std::uint16_t attribute_handle )
269  {
270  if ( args.type != details::attribute_access_type::read )
271  return details::attribute_access_result::write_not_permitted;
272 
273  const std::uint8_t properties[] = {
274  static_cast< std::uint8_t >(
275  ( value_type::has_read_access ? bits( details::gatt_characteristic_properties::read ) : 0 ) |
276  ( value_type::has_write_access ? bits( details::gatt_characteristic_properties::write ) : 0 ) |
277  ( value_type::has_write_without_response ? bits( details::gatt_characteristic_properties::write_without_response ) : 0 ) |
278  ( value_type::has_notification ? bits( details::gatt_characteristic_properties::notify ) : 0 ) |
279  ( value_type::has_indication ? bits( details::gatt_characteristic_properties::indicate ) : 0 ) )
280  };
281 
282  // the Characteristic Value Declaration must follow directly behind this attribute and has, thus the next handle
283  const std::uint8_t value_handle[] = {
284  static_cast< std::uint8_t >( ( attribute_handle +1 ) & 0xff ),
285  static_cast< std::uint8_t >( ( attribute_handle +1 ) >> 8 )
286  };
287 
288  static constexpr auto data_size = sizeof( properties ) + sizeof( value_handle ) + sizeof( uuid::bytes );
289 
290  if ( args.buffer_offset > data_size )
291  return details::attribute_access_result::invalid_offset;
292 
293  details::scattered_read_access( args.buffer_offset, properties, value_handle, uuid::bytes, args.buffer, args.buffer_size );
294 
295  if ( characteristic_or_service_uuid< Options... >::auto_generated_uuid )
296  fixup_auto_uuid( args );
297 
298  args.buffer_size = std::min< std::size_t >( data_size - args.buffer_offset, args.buffer_size );
299 
300  return details::attribute_access_result::success;
301  }
