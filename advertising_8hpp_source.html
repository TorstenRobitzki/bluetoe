49 template < typename Layout >
50 static bool is_valid_scan_request( const read_buffer& receive, const device_address& addr )
51 {
52 static constexpr std::size_t scan_request_size = 2 * address_length;
53 static constexpr std::uint8_t scan_request_code = 0x03;
54
55 const std::uint16_t header = Layout::header( receive );
56
57 bool result = receive.size == Layout::data_channel_pdu_memory_size( scan_request_size )
58 && ( ( header >> 8 ) & 0x3f ) == scan_request_size
59 && ( header & 0x0f ) == scan_request_code;
60
61 if ( result )
62 {
63 const auto body = Layout::body( receive );
64 result = result && std::equal( &body.begin[ address_length ], &body.begin[ 2 * address_length ], addr.begin() );
65 result = result && addr.is_random() == ( ( header & header_rxaddr_field ) != 0 );
66 }
67
68 return result;
69 }

71 template < typename Layout >
72 static bool is_valid_connect_request( const read_buffer& receive, const device_address& addr )
73 {
74 static constexpr std::size_t connect_request_size = 34;
75 static constexpr std::uint8_t connect_request_code = 0x05;
76
77 if ( receive.size != Layout::data_channel_pdu_memory_size( connect_request_size ) )
78 return false;
79
80 const std::uint16_t header = Layout::header( receive );
81 const auto body = Layout::body( receive ).first;
82
83 bool result = ( ( header >> 8 ) & 0x3f ) == connect_request_size
84 && ( header & 0x0f ) == connect_request_code;
85
86 result = result && std::equal( &body[ address_length ], &body[ 2 * address_length ], addr.begin() );
87 result = result && addr.is_random() == ( ( header & header_rxaddr_field ) != 0 );
88
89 return result;
90 }

92 template < typename Layout >
93 static std::size_t fill_empty_advertising_response_data( const device_address& addr, read_buffer adv_response_buffer )
94 {
95 std::uint16_t header = scan_response_pdu_type_code;
96
97 if ( addr.is_random() )
98 header |= header_txaddr_field;
99
100 static constexpr std::size_t empty_ad_size = 2;
101 std::size_t adv_response_size = advertising_pdu_header_size + address_length + empty_ad_size;
102 header |= ( adv_response_size - advertising_pdu_header_size ) << 8;
103
104 const auto body = Layout::body( adv_response_buffer );
105
106 std::copy( addr.begin(), addr.end(), body.first );
107
108 // add aditional empty AD to be visible to Nordic sniffer.
109 // Some stacks do not recognize the response without this empty AD.
110 body.first[ adv_response_size - 2 ] = 0;
111 body.first[ adv_response_size - 1 ] = 0;
112
113 Layout::header( adv_response_buffer, header );
114
115 return adv_response_size;
116 }
