563  template < typename T >
564  inline std::pair< std::uint8_t, T > deserialize( std::size_t write_size, const std::uint8_t* value )
565  {
566  static_assert( CHAR_BIT == 8, "needs porting!" );
567 
568  if ( write_size != sizeof( T ) )
569  return std::pair< std::uint8_t, T >{ error_codes::invalid_attribute_value_length, 0 };
570 
571  T result = 0;
572 
573  for ( const std::uint8_t* v = value + sizeof( T ); v != value; --v )
574  {
575  result = ( result << 8 ) | *( v - 1 );
576  }
577 
578  return std::pair< std::uint8_t, T >{ error_codes::success, result };
579  }
