406  template < typename T >
407  inline std::pair< std::uint8_t, T > deserialize( std::size_t write_size, const std::uint8_t* value )
408  {
409  static_assert( CHAR_BIT == 8, "needs porting!" );
410 
411  if ( write_size != sizeof( T ) )
412  return std::pair< std::uint8_t, T >{ error_codes::invalid_attribute_value_length, 0 };
413 
414  T result = 0;
415 
416  for ( const std::uint8_t* v = value + sizeof( T ); v != value; --v )
417  {
418  result = ( result << 8 ) | *( v - 1 );
419  }
420 
421  return std::pair< std::uint8_t, T >{ error_codes::success, result };
422  }
