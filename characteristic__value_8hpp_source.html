618  template < typename T >
619  inline std::pair< std::uint8_t, T > deserialize( std::size_t write_size, const std::uint8_t* value )
620  {
621  static_assert( CHAR_BIT == 8, "needs porting!" );
622 
623  if ( write_size != sizeof( T ) )
624  return std::pair< std::uint8_t, T >{ error_codes::invalid_attribute_value_length, 0 };
625 
626  T result = 0;
627 
628  for ( const std::uint8_t* v = value + sizeof( T ); v != value; --v )
629  {
630  result = ( result << 8 ) | *( v - 1 );
631  }
632 
633  return std::pair< std::uint8_t, T >{ error_codes::success, result };
634  }
