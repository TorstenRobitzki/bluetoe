373  template < typename T >
374  inline std::pair< std::uint8_t, T > deserialize( std::size_t write_size, const std::uint8_t* value )
375  {
376  static_assert( CHAR_BIT == 8, "needs porting!" );
377 
378  if ( write_size != sizeof( T ) )
379  return std::pair< std::uint8_t, T >{ error_codes::invalid_attribute_value_length, 0 };
380 
381  T result = 0;
382 
383  for ( const std::uint8_t* v = value + sizeof( T ); v != value; --v )
384  {
385  result = ( result << 8 ) | *( v - 1 );
386  }
387 
388  return std::pair< std::uint8_t, T >{ error_codes::success, result };
389  }
