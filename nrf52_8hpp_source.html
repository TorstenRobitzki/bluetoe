356 bluetoe::link_layer::delta_time when,
357 const bluetoe::link_layer::read_buffer& receive )
358 {
359 assert( !adv_received_ );
360 assert( !adv_timeout_ );
361 assert( state_ == state::idle );
362 assert( receive.buffer && receive.size >= 2u );
363
364 bluetoe::link_layer::write_buffer advertising = advertising_data;
365 advertising.size = std::min< std::uint8_t >( advertising.size, maximum_advertising_pdu_size );
366

385 link_layer::delta_time schedule_connection_event(
386 unsigned channel,
387 bluetoe::link_layer::delta_time start_receive,
388 bluetoe::link_layer::delta_time end_receive,
389 bluetoe::link_layer::delta_time connection_interval )
390 {
391 assert( !end_evt_ );
392 assert( state_ == state::idle );
393 assert( start_receive < end_receive );
394
395 // in case of a call to nrf_flash_memory_access_end, evt_timeout_ will be set
396 // and needs to be consumed by the timeout handler
397 if ( evt_timeout_ )
398 return connection_interval;
399
400 // Stop all interrupts so that the calculation, that enough CPU time is available to setup everything, will not
401 // be disturbed by any interrupt.
402 lock_guard lock;
403
404 const std::uint32_t start_event = start_receive.usec();
405 // TODO: 500: must depend on receive size. This probably would depend on the maximum
406 // LL PDU size that would be expected.
407 const std::uint32_t end_event = end_receive.usec() + 500;
408
409 const auto now = Hardware::now();
410 if ( now + start_event_safety_margin_us > start_event )
411 {
412 evt_timeout_ = true;
413 return connection_interval;
414 }
415
416 receive_buffer_ = receive_buffer();
417 state_ = state::evt_wait_connect;
418
419 Hardware::configure_radio_channel( channel );
420 Hardware::configure_receive_train( receive_buffer_ );
421
422 Hardware::schedule_connection_event_timer( start_event, end_event, hfxo_startup_value );
423 low_frequency_clock_t::stop_high_frequency_crystal_oscilator();
424
425 return link_layer::delta_time( connection_interval.usec() - now );
426 }

466 void run()
467 {
468 while ( !adv_received_ && !adv_timeout_ && !evt_timeout_ && !end_evt_ && wake_up_ == 0 )
469 __WFI();
470
471 // For every event, but the wakeup request, the radio should be in an idle state
472 // because it's up to the event handler to defined what should happen next
473 if ( adv_received_ )
474 {
475 assert( state_ == state::idle );
476 assert( reinterpret_cast< std::uint8_t* >( NRF_RADIO->PACKETPTR ) == receive_buffer_.buffer );
477
478 receive_buffer_.size = std::min< std::size_t >(
479 receive_buffer_.size,
480 ( receive_buffer_.buffer[ 1 ] & 0x3f ) + 2 + Hardware::pdu_gap_required_by_encryption()
481 );
482 adv_received_ = false;
483
484 static_cast< CallBacks* >( this )->adv_received( receive_buffer_ );
485 }
486
487 if ( adv_timeout_ )
488 {
489 assert( state_ == state::idle );
490 adv_timeout_ = false;
491 static_cast< CallBacks* >( this )->adv_timeout();
492 }
493
494 if ( evt_timeout_ )
495 {
496 assert( state_ == state::idle );
497 evt_timeout_ = false;
498 static_cast< CallBacks* >( this )->timeout();
499 }
500
501 if ( end_evt_ )
502 {
503 assert( state_ == state::idle );
504 end_evt_ = false;
505 static_cast< CallBacks* >( this )->end_event( events_ );
506
507 events_ = link_layer::connection_event_events();
508 }
509
510 if ( request_event_cancelation_ )
511 {
512 request_event_cancelation_ = false;
513 static_cast< CallBacks* >( this )->try_event_cancelation();
514 }
515
516 if ( wake_up_ )
517 {
518 --wake_up_;
519 }
520 }

523 {
524 ++wake_up_;
525 }

615 void radio_interrupt_handler()
616 {
617 if ( state_ == state::adv_transmitting )
618 {
619 // The timeout timer was already set with the start of the advertising
620 // Configure Radio to receive and then switch to transmitting
621 Hardware::configure_receive_train( receive_buffer_ );
622 state_ = state::adv_receiving;
623 }
624 else if ( state_ == state::adv_receiving )
625 {
626 bool valid_anchor, valid_pdu, valid_crc;
627 std::tie( valid_anchor, valid_pdu, valid_crc ) = Hardware::received_pdu();
628
629 if ( valid_anchor && valid_pdu && valid_crc )
630 {
631 Hardware::stop_timeout_timer();
632
633 if ( is_valid_scan_request() )
634 {
635 Hardware::configure_final_transmit( response_data_ );
636 state_ = state::adv_transmitting_response;
637
638 return;
639 }
640
641 adv_received_ = true;
642
643 // In case of an connection request, the anchor is exactly the end
644 // of the request
645 Hardware::store_timer_anchor( 0 );
646 }
647 else
648 {
649 adv_timeout_ = true;
650 }
651
652 Hardware::stop_radio();
653
654 state_ = state::idle;
655 }
656 else if ( state_ == state::adv_transmitting_response )
657 {
658 Hardware::stop_radio();
659
660 state_ = state::idle;
661 adv_timeout_ = true;
662 }
663 else if ( state_ == state::evt_wait_connect )
664 {
665 bool valid_anchor, valid_pdu, valid_crc;
666 std::tie( valid_anchor, valid_pdu, valid_crc ) = Hardware::received_pdu();
667
668 if ( valid_anchor && ( valid_pdu || valid_crc ) )
669 {
670 // switch to transmission
671 const auto trans = ( receive_buffer_.buffer == &empty_receive_[ 0 ] || !valid_crc )
672 ? this->next_transmit()
673 : ( valid_pdu
674 ? this->received( receive_buffer_ )
675 : this->acknowledge( receive_buffer_ ) );
676
677 // TODO: Hack to disable the more data flag, because this radio implementation is currently
678 // not able to do this (but it should be possible with the given hardware).
679 // Issue: #75 More Data not working
680 const_cast< std::uint8_t* >( trans.buffer )[ 0 ] = trans.buffer[ 0 ] & ~more_data_flag;
681
682 if ( trans.buffer[ 1 ] != 0 )
683 events_.last_transmitted_not_empty = true;
684
685 Hardware::configure_final_transmit( trans );
686 state_ = state::evt_transmiting_closing;
687 }
688 else
689 {
690 Hardware::stop_radio();
691 events_.error_occured = true;
692 evt_timeout_ = true;
693 state_ = state::idle;
694 }
695 }
696 else if ( state_ == state::evt_transmiting_closing )
697 {
698 // TODO: Couldn't we just capture the time at the start of the PDU?
699 // the timer was captured with the end event; the anchor is the start of the receiving.
700 // Additional to the ll PDU length there are 1 byte preamble, 4 byte access address, 2 byte LL header and 3 byte crc.
701 // Issue: #76 Taking Anchor from End of PDU
702 static constexpr std::size_t ll_pdu_overhead = 1 + 4 + 2 + 3;
703 const int total_pdu_length = ( receive_buffer_.buffer[ 1 ] + ll_pdu_overhead ) * 8;
704
705 // TODO: Anchor must also be taken, if PDU has a CRC error
706 // Issue: #76 Taking Anchor from End of PDU
707 Hardware::store_timer_anchor( -total_pdu_length );
708
709 if ( receive_buffer_.buffer[ 1 ] != 0 )
710 events_.last_received_not_empty = true;
711
712 if ( receive_buffer_.buffer[ 0 ] & more_data_flag )
713 events_.last_received_had_more_data = true;
714
715 Hardware::stop_radio();
716 state_ = state::idle;
717 end_evt_ = true;
718 }
719 else
720 {
721 assert(!"Invalid state");
722 }

725 bool is_valid_scan_request() const
726 {
727 static constexpr std::uint8_t scan_request_size = 12;
728 static constexpr std::uint8_t scan_request_pdu_type = 0x03;
729 static constexpr std::uint8_t pdu_type_mask = 0x0F;
730 static constexpr int pdu_header_size = 2;
731 static constexpr int addr_size = 6;
732 static constexpr std::uint8_t tx_add_mask = 0x40;
733 static constexpr std::uint8_t rx_add_mask = 0x80;
734
735 // the advertising type does not expect a scan request
736 if ( !response_data_.buffer )
737 return false;
738
739 if ( Hardware::resolving_address_invalid() )
740 return true;
741
742 if ( receive_buffer_.buffer[ 1 ] != scan_request_size )
743 return false;
744
745 if ( ( receive_buffer_.buffer[ 0 ] & pdu_type_mask ) != scan_request_pdu_type )
746 return false;
747
748 const int pdu_gap = Hardware::pdu_gap_required_by_encryption();
749
750 if ( !std::equal( &receive_buffer_.buffer[ pdu_header_size + addr_size + pdu_gap ], &receive_buffer_.buffer[ pdu_header_size + 2 * addr_size + pdu_gap ],
751 &response_data_.buffer[ pdu_header_size + pdu_gap ] ) )
752 return false;
753
754 // in the scan request, the randomness is stored in RxAdd, in the scan response, it's stored in
755 // TxAdd.
756 const bool scanner_addres_is_random = response_data_.buffer[ 0 ] & tx_add_mask;
757 if ( !static_cast< bool >( receive_buffer_.buffer[ 0 ] & rx_add_mask ) == scanner_addres_is_random )
758 return false;
759
760 const link_layer::device_address scanner( &receive_buffer_.buffer[ pdu_header_size + pdu_gap ], scanner_addres_is_random );
761
762 return static_cast< const CallBacks* >( this )->is_scan_request_in_filter( scanner );
763 }
