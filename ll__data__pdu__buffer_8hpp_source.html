431  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
432  write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::set_next_expected_sequence_number( read_buffer buf ) const
433  {
434  // insert the next expected sequence for every attempt to send the PDU, because it could be that
435  // the slave is able to receive data, while the master is not able to.
436  std::size_t header = layout::header( buf );
437 
438  header = next_expected_sequence_number_
439  ? ( header | nesn_flag )
440  : ( header & ~nesn_flag );
441

447  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
448  write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::next_transmit()
449  {
450  const read_buffer next = transmit_buffer_.next_end();
451 
452  if ( next_empty_ )
453  {
454  // if an empty buffer have to be resend, flag that there is more data
455  if ( next.size )
456  {
457  const std::uint16_t header = layout::header( next ) | more_data_flag;
458  layout::header( next, header );
459  }
460 
461  return set_next_expected_sequence_number( read_buffer{ &empty_[ 0 ], sizeof( empty_ ) } );
462  }
463  else if ( next.size == 0 )
464  {
465  // we created an PDU, so it have to have a new sequnce number
466  const std::uint16_t header = sequence_number_
467  ? sn_flag + ll_empty_id
468  : ll_empty_id;
469 
470  layout::header( empty_, header );
471  next_empty_ = true;
472 
473  // keep sequence number of empty in mind and increment sequence_number_
474  empty_sequence_number_ = sequence_number_;
475  sequence_number_ = !sequence_number_;
476 
477  return set_next_expected_sequence_number( read_buffer{ &empty_[ 0 ], sizeof( empty_ ) } );
478  }
479 
480  if ( transmit_buffer_.more_than_one() )
481  layout::header( next, layout::header( next ) | more_data_flag );
482 
483  return set_next_expected_sequence_number( next );
484  }

486  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
487  void ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::acknowledge( bool nesn )
488  {
489  if ( next_empty_ )
490  {
491  if ( empty_sequence_number_ != nesn )
492  next_empty_ = false;
493  }
494  else
495  {
496  const read_buffer next = transmit_buffer_.next_end();
497 
498  // the transmit buffer could be empty if we receive without sending prior. That happens during testing
499  if ( next.empty() )
500  return;
501 
502  const std::uint16_t header = layout::header( next );
503  if ( static_cast< bool >( header & sn_flag ) != nesn )
504  {
505  transmit_buffer_.pop_end( transmit_buffer() );
506  static_cast< Radio* >( this )->increment_transmit_packet_counter();
507  }
508  }
