414 template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
415 void ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::commit_transmit_buffer( read_buffer pdu )
416 {
417 static constexpr std::uint8_t header_rfu_mask = 0xe0;
418 static_cast< void >( header_rfu_mask );
419
420 // make sure, no NFU bits are set
421 std::uint16_t header = layout::header( pdu );
422 assert( ( header & header_rfu_mask ) == 0 );
423
424 typename Radio::lock_guard lock;
425
426 // add sequence number
427 if ( sequence_number_ )
428 {
429 header |= sn_flag;
430 layout::header( pdu, header );
431 }
432
433 sequence_number_ = !sequence_number_;
434
435 transmit_buffer_.push_front( transmit_buffer(), pdu );
436 }

438 template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
439 write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::set_next_expected_sequence_number( read_buffer buf ) const
440 {
441 // insert the next expected sequence for every attempt to send the PDU, because it could be that
442 // the slave is able to receive data, while the master is not able to.
443 std::size_t header = layout::header( buf );
444
445 header = next_expected_sequence_number_
446 ? ( header | nesn_flag )
447 : ( header & ~nesn_flag );
448
449 layout::header( buf, header );
450
451 return write_buffer( buf );
452 }

454 template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
455 write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::next_transmit()
456 {
457 const read_buffer next = transmit_buffer_.next_end();
458
459 if ( next_empty_ )
460 {
461 // if an empty buffer have to be resend, flag that there is more data
462 if ( next.size )
463 {
464 const std::uint16_t header = layout::header( next ) | more_data_flag;
465 layout::header( next, header );
466 }
467
468 return set_next_expected_sequence_number( read_buffer{ &empty_[ 0 ], sizeof( empty_ ) } );
469 }
470 else if ( next.size == 0 )
471 {
472 // we created an PDU, so it have to have a new sequnce number
473 const std::uint16_t header = sequence_number_
474 ? sn_flag + ll_empty_id
475 : ll_empty_id;
476
477 layout::header( empty_, header );
478 next_empty_ = true;
479
480 // keep sequence number of empty in mind and increment sequence_number_
481 empty_sequence_number_ = sequence_number_;
482 sequence_number_ = !sequence_number_;
483
484 return set_next_expected_sequence_number( read_buffer{ &empty_[ 0 ], sizeof( empty_ ) } );
485 }
486
487 if ( transmit_buffer_.more_than_one() )
488 layout::header( next, layout::header( next ) | more_data_flag );
489
490 return set_next_expected_sequence_number( next );
491 }

493 template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
494 void ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::acknowledge( bool nesn )
495 {
496 if ( next_empty_ )
497 {
498 if ( empty_sequence_number_ != nesn )
499 next_empty_ = false;
500 }
501 else
502 {
503 const read_buffer next = transmit_buffer_.next_end();
504
505 // the transmit buffer could be empty if we receive without sending prior. That happens during testing
506 if ( next.empty() )
507 return;
508
509 const std::uint16_t header = layout::header( next );
510 if ( static_cast< bool >( header & sn_flag ) != nesn )
511 {
512 transmit_buffer_.pop_end( transmit_buffer() );
513 static_cast< Radio* >( this )->increment_transmit_packet_counter();
514 }
515 }
516 }

546 template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
547 write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::received( read_buffer pdu )
548 {
549 const std::uint16_t header = layout::header( pdu );
550
551 // invalid LLID
552 if ( ( header & 0x3 ) != 0 )
553 {
554 acknowledge( header & nesn_flag );
555
556 // resent PDU?
557 if ( static_cast< bool >( header & sn_flag ) == next_expected_sequence_number_ )
558 {
559 next_expected_sequence_number_ = !next_expected_sequence_number_;
560
561 if ( ( header & 0xff00 ) != 0 )
562 {
563 receive_buffer_.push_front( receive_buffer(), pdu );
564 static_cast< Radio* >( this )->increment_receive_packet_counter();
565 }
566 }
567 }
568
569 return next_transmit();
570 }
