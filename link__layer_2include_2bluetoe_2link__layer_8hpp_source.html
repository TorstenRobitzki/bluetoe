130 bool handle_encryption_pdus( std::uint8_t opcode, std::uint8_t size, const write_buffer& pdu, read_buffer write, bool& commit )
131 {
132 using namespace ::bluetoe::details;
133
134 using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
135
136 bool encryption_changed = false;
137
138 if ( opcode == LinkLayer::LL_ENC_REQ && size == 23 )
139 {
140 encryption_in_progress_ = true;
141 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1 + 8 + 4, LinkLayer::LL_ENC_RSP } );
142
143 const std::uint8_t* const pdu_body = layout_t::body( pdu ).first;
144
145 const std::uint64_t rand = read_64bit( pdu_body +1 );
146 const std::uint16_t ediv = read_16bit( pdu_body +9 );
147 const std::uint64_t skdm = read_64bit( pdu_body +11 );
148 const std::uint32_t ivm = read_32bit( pdu_body +19 );
149 std::uint64_t skds = 0;
150 std::uint32_t ivs = 0;
151
152 bluetoe::details::uint128_t key;
153 std::tie( has_key_, key ) = that().connection_data_.find_key( ediv, rand );
154
155 // setup encryption
156 std::tie( skds, ivs ) = that().setup_encryption( key, skdm, ivm );
157
158 std::uint8_t* write_body = layout_t::body( write ).first;
159 bluetoe::details::write_64bit( &write_body[ 1 ], skds );
160 bluetoe::details::write_32bit( &write_body[ 9 ], ivs );
161 }
162 else if ( opcode == LinkLayer::LL_START_ENC_RSP && size == 1 )
163 {
164 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_RSP } );
165 that().start_transmit_encrypted();
166 encryption_changed = that().connection_data_.is_encrypted( true );
167
168 if ( encryption_changed )
169 that().connection_data_.restore_bonded_cccds( that().connection_data_ );
170
171 }
172 else if ( opcode == LinkLayer::LL_PAUSE_ENC_REQ && size == 1 )
173 {
174 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_PAUSE_ENC_RSP } );
175 that().stop_receive_encrypted();
176 encryption_changed = that().connection_data_.is_encrypted( false );
177 }
178 else if ( opcode == LinkLayer::LL_PAUSE_ENC_RSP && size == 1 )
179 {
180 that().stop_transmit_encrypted();
181 encryption_changed = that().connection_data_.is_encrypted( false );
182
183 commit = false;
184 }
185 else
186 {
187 return false;
188 }
189
190 if ( encryption_changed )
191 {
192 that().connection_data_.pairing_status(that().connection_data_.local_device_pairing_status());
193 that().connection_changed( that().details(), that().connection_data_, static_cast< typename LinkLayer::radio_t& >( that() ) );
194 }
195
196 return true;
197 }
198
199 void transmit_pending_security_pdus()
200 {
201 using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
202
203 if ( !encryption_in_progress_ )
204 return;
205
206 auto out_buffer = that().allocate_ll_transmit_buffer( LinkLayer::maximum_ll_payload_size );
207 if ( out_buffer.empty() )
208 return;
209
210 if ( has_key_ )
211 {
212 fill< layout_t >( out_buffer, {
213 LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_REQ } );
214
215 that().start_receive_encrypted();
216 that().commit_ll_transmit_buffer( out_buffer );
217 }
218 else
219 {
220 that().reject( LinkLayer::LL_ENC_REQ, LinkLayer::err_pin_or_key_missing, out_buffer );
221 that().commit_ll_transmit_buffer( out_buffer );
222 }
223
224 encryption_in_progress_ = false;
225 }
226
227 void reset_encryption()
228 {
229 that().connection_data_.is_encrypted( false );
230 that().stop_receive_encrypted();
231 that().stop_transmit_encrypted();
232 }
233
234 private:
235 bool has_key_;
236 bool encryption_in_progress_;
237 };

269 template < class LL >
270 bool handle_phy_request( std::uint8_t opcode, std::uint8_t size, const write_buffer& pdu, read_buffer& write, LL& link_layer, bool& commit )
271 {
272 assert( link_layer.defered_ll_control_pdu_.buffer == nullptr );
273
274 using layout_t = typename pdu_layout_by_radio< typename LL::radio_t >::pdu_layout;
275
276 if ( opcode == LL::LL_PHY_REQ && size == 3 )
277 {
278 fill< layout_t >( write, {
279 LL::ll_control_pdu_code, 3,
280 LL::LL_PHY_RSP,
281 phy_ll_encoding::le_1m_phy | phy_ll_encoding::le_2m_phy,
282 phy_ll_encoding::le_1m_phy | phy_ll_encoding::le_2m_phy } );
283
284 return true;
285 }
286
287 if ( opcode == LL::LL_PHY_UPDATE_IND && size == 5 )
288 {
289 const std::uint8_t* const pdu_body = layout_t::body( pdu ).first;
290
291 const std::uint8_t c_to_p = pdu_body[ 1 ];
292 const std::uint8_t p_to_c = pdu_body[ 2 ];
293
294 if ( !valid_phy_encoding( c_to_p ) || !valid_phy_encoding( p_to_c ) )
295 return false;
296
297 commit = false;
298
299 if ( c_to_p == details::phy_ll_encoding::le_unchanged_coding
300 && p_to_c == details::phy_ll_encoding::le_unchanged_coding )
301 return true;
302
303 link_layer.defered_ll_control_pdu_ = pdu;
304 link_layer.defered_conn_event_counter_ = ::bluetoe::details::read_16bit( pdu_body + 3 );
305
306 return true;
307 }
308
309 return false;
310 }

312 template < class LL >
313 bool handle_pending_phy_request( std::uint8_t opcode, LL& link_layer )
314 {
315 assert( link_layer.defered_ll_control_pdu_.buffer );
316
317 using layout_t = typename pdu_layout_by_radio< typename LL::radio_t >::pdu_layout;
318
319 if ( opcode == LL::LL_PHY_UPDATE_IND )
320 {
321 const std::uint8_t* const pdu_body = layout_t::body( link_layer.defered_ll_control_pdu_ ).first;
322
323 const auto c_to_p = static_cast< phy_ll_encoding::phy_ll_encoding_t >( pdu_body[ 1 ] );
324 const auto p_to_c = static_cast< phy_ll_encoding::phy_ll_encoding_t >( pdu_body[ 2 ] );
325 link_layer.defered_ll_control_pdu_ = { nullptr, 0 };
326 link_layer.radio_set_phy( c_to_p, p_to_c );
327
328 return true;
329 }
330
331 return false;
332 }

889 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
890 void link_layer< Server, ScheduledRadio, Options... >::end_event( connection_event_events evts )
891 {
892 pending_event_ = false;
893
894 assert( state_ == state::connecting || state_ == state::connected || state_ == state::disconnecting || state_ == state::connection_changed );
895
896 if ( state_ == state::connecting || restart_user_timer_requested_ )
897 {
898 this->synchronized_connection_event_callback_new_connection( connection_interval_ );
899 restart_user_timer_requested_ = false;
900 }
901
902 if ( state_ == state::connecting )
903 {
904 this->connection_established( details(), connection_data_, static_cast< radio_t& >( *this ) );
905 }
906 else if ( state_ == state::connection_changed )
907 {
908 this->synchronized_connection_event_callback_connection_changed( connection_interval_ );
909 }
910
911 if ( state_ != state::disconnecting )
912 {
913 state_ = state::connected;
914 transmit_window_size_ = delta_time();
915 }
916
917 /*
918 * L2CAP and pending LL PDUs are handle first, to get the information whether there is
919 * data to be send. This implies that the connEventCount is not updated at this point
920 * and has to be offset by 1 to see if there is a pending instant at this connection
921 * event.
922 */
923 if ( handle_received_data() == ll_result::disconnect || send_control_pdus() == ll_result::disconnect )
924 {
925 force_disconnect();
926 }
927 else
928 {
929 this->transmit_pending_security_pdus();
930
931 const std::pair< bool, std::uint16_t > pending_instant = { !defered_ll_control_pdu_.empty(), defered_conn_event_counter_ };
932
933 evts.pending_outgoing_data = evts.pending_outgoing_data || this->pending_outgoing_data_available();
934 this->plan_next_connection_event(
935 peripheral_latency_, evts, connection_interval_, pending_instant );
936
937 // Handle pending LL control PDUs that will affect the _next_ connection event
938 if ( handle_pending_ll_control( this->connection_event_counter() ) == ll_result::disconnect )
939 {
940 force_disconnect();
941 }
942 else
943 {
944 const delta_time time_till_next_event = setup_next_connection_event();
945 connection_event_callback::call_connection_event_callback( time_till_next_event );
946 }
947
948 }
949
950 if ( state_ == state::connected || state_ == state::connecting )
951 {
952 transmit_pending_control_pdus();
953 this->transmit_pending_l2cap_output( connection_data_ );
954 }
955
956 this->template handle_connection_events< link_layer< Server, ScheduledRadio, Options... > >();
957 }

1033 {

1035
1036 delta_time window_start;
1037 delta_time window_end;
1038

1040
1041 // optimization to calculate the deviation only once for the symetrical case
1042 if ( !transmit_window_size_.zero() )
1043 {
1044 window_start = time_since_last_event + transmit_window_offset_;
1045 window_end = window_start + transmit_window_size_;

1047 window_start -= window_start.ppm( cumulated_sleep_clock_accuracy_ );
1048 window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );

1050 else
1051 {
1052 const delta_time window_size = time_since_last_event.ppm( cumulated_sleep_clock_accuracy_ );
1053
1054 window_start = time_since_last_event - window_size;
1055 window_end = time_since_last_event + window_size;
1056 }
1057
1058 return this->schedule_connection_event(
1059 channels_.data_channel( this->current_channel_index() ),
1060 window_start,
1061 window_end,
1062 connection_interval_ );

1307 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
1308 typename link_layer< Server, ScheduledRadio, Options... >::ll_result link_layer< Server, ScheduledRadio, Options... >::handle_ll_control_data( const write_buffer& pdu, read_buffer write )
1309 {
1310 ll_result result = ll_result::go_ahead;
1311 bool commit = true;
1312
1313 assert( write.size >= radio_t::min_buffer_size );
1314
1315 const std::uint8_t* const body = layout_t::body( pdu ).first;
1316 const std::uint16_t header = layout_t::header( pdu );
1317
1318 if ( ( header & 0x3 ) == ll_control_pdu_code )
1319 {
1320 const std::uint8_t size = header >> 8;
1321 const std::uint8_t opcode = size > 0 ? *body : 0xff;
1322
1323 if ( opcode == LL_CONNECTION_UPDATE_REQ && size == 12 )
1324 {
1325 defered_conn_event_counter_ = bluetoe::details::read_16bit( &body[ 10 ] );
1326 commit = false;
