124 bool handle_encryption_pdus( std::uint8_t opcode, std::uint8_t size, const write_buffer& pdu, read_buffer write )
125 {
126 using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
127
128 bool encryption_changed = false;
129 bool result = true;
130
131 if ( opcode == LinkLayer::LL_ENC_REQ && size == 23 )
132 {
133 encryption_in_progress_ = true;
134 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1 + 8 + 4, LinkLayer::LL_ENC_RSP } );
135
136 const std::uint8_t* const pdu_body = layout_t::body( pdu ).first;
137
138 const std::uint64_t rand = LinkLayer::read_64( pdu_body +1 );
139 const std::uint16_t ediv = LinkLayer::read_16( pdu_body +9 );
140 const std::uint64_t skdm = LinkLayer::read_64( pdu_body +11 );
141 const std::uint32_t ivm = LinkLayer::read_32( pdu_body +19 );
142 std::uint64_t skds = 0;
143 std::uint32_t ivs = 0;
144
145 bluetoe::details::uint128_t key;
146 std::tie( has_key_, key ) = that().connection_details_.find_key( ediv, rand );
147
148 // setup encryption
149 std::tie( skds, ivs ) = that().setup_encryption( key, skdm, ivm );
150
151 std::uint8_t* write_body = layout_t::body( write ).first;
152 bluetoe::details::write_64bit( &write_body[ 1 ], skds );
153 bluetoe::details::write_32bit( &write_body[ 9 ], ivs );
154 }
155 else if ( opcode == LinkLayer::LL_START_ENC_RSP && size == 1 )
156 {
157 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_RSP } );
158 that().start_transmit_encrypted();
159 encryption_changed = that().connection_details_.is_encrypted( true );
160 }
161 else if ( opcode == LinkLayer::LL_PAUSE_ENC_REQ && size == 1 )
162 {
163 fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_PAUSE_ENC_RSP } );
164 that().stop_receive_encrypted();
165 encryption_changed = that().connection_details_.is_encrypted( false );
166 }
167 else if ( opcode == LinkLayer::LL_PAUSE_ENC_RSP && size == 1 )
168 {
169 that().stop_transmit_encrypted();
170 encryption_changed = that().connection_details_.is_encrypted( false );
171
172 result = false;
173 }
174 else
175 {
176 result = false;
177 }
178
179 if ( encryption_changed )
180 that().connection_changed( that().details(), that().connection_details_, static_cast< typename LinkLayer::radio_t& >( that() ) );
181
182 return result;
183 }
184
185 void transmit_pending_security_pdus()
186 {
187 using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
188
189 if ( !encryption_in_progress_ )
190 return;
191
192 auto out_buffer = that().allocate_ll_transmit_buffer( LinkLayer::maximum_ll_payload_size );
193 if ( out_buffer.empty() )
194 return;
195
196 if ( has_key_ )
197 {
198 fill< layout_t >( out_buffer, {
199 LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_REQ } );
200
201 that().start_receive_encrypted();
202 that().commit_ll_transmit_buffer( out_buffer );
203 }
204 else
205 {
206 that().reject( LinkLayer::LL_ENC_REQ, LinkLayer::err_pin_or_key_missing, out_buffer );
207 that().commit_ll_transmit_buffer( out_buffer );
208 }
209
210 encryption_in_progress_ = false;
211 }
212
213 void reset_encryption()
214 {
215 that().connection_details_.is_encrypted( false );
216 that().stop_receive_encrypted();
217 that().stop_transmit_encrypted();
218 }
219
220 private:
221 bool has_key_;
222 bool encryption_in_progress_;
223 };
224 };

604 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
605 void link_layer< Server, ScheduledRadio, Options... >::adv_received( const read_buffer& receive )
606 {
607 assert( state_ == state::advertising );
608
609 device_address remote_address;
610 const bool connection_request_received = this->handle_adv_receive( receive, remote_address );
611
612 if ( connection_request_received )
613 {
614 const std::uint8_t* const body = layout_t::body( receive ).first;
615
616 if ( channels_.reset( &body[ 28 ], body[ 33 ] & 0x1f )
617 && parse_timing_parameters_from_connect_request( body ) )
618 {
619 state_ = state::connecting;
620 current_channel_index_ = 0;
621 conn_event_counter_ = 0;
622 cumulated_sleep_clock_accuracy_ = sleep_clock_accuracy( body ) + device_sleep_clock_accuracy::accuracy_ppm;
623 used_features_ = supported_features;
624 connection_parameters_request_pending_ = false;
625 connection_parameters_request_running_ = false;
626 time_till_next_event_ = delta_time();
627
628 this->set_access_address_and_crc_init( read_32( &body[ 12 ] ), read_24( &body[ 16 ] ) );
629
630 const delta_time window_start = transmit_window_offset_ - transmit_window_offset_.ppm( cumulated_sleep_clock_accuracy_ );
631 delta_time window_end = transmit_window_offset_ + transmit_window_size_;
632
633 window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
634
635 this->reset();
636 this->schedule_connection_event(
637 channels_.data_channel( current_channel_index_ ),
638 window_start,
639 window_end,
640 connection_interval_ );
641
642 this->connection_request( connection_addresses( address_, remote_address ) );
643 this->handle_stop_advertising();
644
645 connection_details_ = connection_details_t( std::size_t{ details::mtu_size< Options... >::mtu } );
646 connection_details_.remote_connection_created( remote_address );
647 }
648 }

659 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
660 void link_layer< Server, ScheduledRadio, Options... >::timeout()
661 {
662 assert( state_ == state::connecting || state_ == state::connected || state_ == state::connection_update || state_ == state::disconnecting );
663
664 if ( time_till_next_event_ <= connection_timeout_
665 && !( state_ == state::connecting && time_till_next_event_ >= ( num_windows_til_timeout - 1 ) * connection_interval_ ) )
666 {
667 time_till_next_event_ += connection_interval_;
668
669 current_channel_index_ = ( current_channel_index_ + 1 ) % first_advertising_channel;
670 ++conn_event_counter_;
671
672 if ( handle_pending_ll_control() == ll_result::disconnect )
673 {
674 force_disconnect();
675 }
676 else
677 {
678 wait_for_connection_event();
679 }
680 }
681 else
682 {
683 force_disconnect();
684 }
685 }

687 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
688 void link_layer< Server, ScheduledRadio, Options... >::end_event()
689 {
690 assert( state_ == state::connecting || state_ == state::connected || state_ == state::connection_update || state_ == state::disconnecting );
691
692 time_till_next_event_ = connection_interval_;
693
694 if ( state_ == state::connecting )
695 {
696 this->connection_established( details(), connection_details_, static_cast< radio_t& >( *this ) );
697 }
698
699 if ( state_ != state::disconnecting )
700 {
701 state_ = state::connected;
702 transmit_window_size_ = delta_time();
703 }
704
705 current_channel_index_ = ( current_channel_index_ + 1 ) % first_advertising_channel;
706 ++conn_event_counter_;
707
708 if ( handle_received_data() == ll_result::disconnect || send_control_pdus() == ll_result::disconnect )
709 {
710 force_disconnect();
711 }
712 else
713 {
714 this->transmit_pending_security_pdus();
715 wait_for_connection_event();
716 }
717 }

755 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
756 void link_layer< Server, ScheduledRadio, Options... >::wait_for_connection_event()
757 {
758 delta_time window_start;
759 delta_time window_end;
760
761 // optimization to calculate the deviation only once for the symetrical case
762 if ( transmit_window_size_.zero() )
763 {
764 const delta_time window_size = time_till_next_event_.ppm( cumulated_sleep_clock_accuracy_ );
765
766 window_start = time_till_next_event_ - window_size;
767 window_end = time_till_next_event_ + window_size;
768 }
769 else
770 {
771 window_start = time_till_next_event_ + transmit_window_offset_;
772 window_end = window_start + transmit_window_size_;
773
774 window_start -= window_start.ppm( cumulated_sleep_clock_accuracy_ );
775 window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
776 }
777
778 const delta_time time_till_next_event = this->schedule_connection_event(
779 channels_.data_channel( current_channel_index_ ),
780 window_start,
781 window_end,
782 connection_interval_ );
783
784 // Do not call the connection callback, if the last connection event timed out, as this could be an
785 // indication for the callback constantly consuming too much CPU time
786 if ( time_till_next_event_ == connection_interval_ )
787 {
788 connection_event_callback::call_connection_event_callback( time_till_next_event );
789 }
790 }

792 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
793 void link_layer< Server, ScheduledRadio, Options... >::transmit_notifications()
794 {
795 // first check if we have memory to transmit the message, or otherwise notifications would get lost
796 auto out_buffer = this->allocate_l2cap_transmit_buffer( connection_details_.negotiated_mtu() );
797
798 if ( out_buffer.empty() )
799 return;
800
801 const auto notification = connection_details_.dequeue_indication_or_confirmation();
802
803 if ( notification.first != connection_details_t::entry_type::empty )
804 {
805 std::size_t out_size = out_buffer.size - all_header_size - layout_t::data_channel_pdu_memory_size( 0 );
806 std::uint8_t* out_body = layout_t::body( out_buffer ).first;
807
808 if ( notification.first == connection_details_t::entry_type::notification )
809 {
810 server_->notification_output(
811 &out_body[ l2cap_header_size ],
812 out_size,
813 connection_details_,
814 notification.second
815 );
816 }
817 else
818 {
819 server_->indication_output(
820 &out_body[ l2cap_header_size ],
821 out_size,
822 connection_details_,
823 notification.second
824 );
825
826 // if no output is generate, confirm the indication, or we will wait for ever
827 if ( out_size == 0 )
828 connection_details_.indication_confirmed();
829
830 }
831
832 if ( out_size )

834 fill< layout_t >( out_buffer, {
835 lld_data_pdu_code,
836 static_cast< std::uint8_t >( out_size + l2cap_header_size ),
837 static_cast< std::uint8_t >( out_size & 0xff ),
838 static_cast< std::uint8_t >( ( out_size & 0xff00 ) >> 8 ),
839 static_cast< std::uint8_t >( l2cap_att_channel ),
840 static_cast< std::uint8_t >( l2cap_att_channel >> 8 ) } );

842 this->commit_l2cap_transmit_buffer( out_buffer );

994 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >

996 {

998 static constexpr delta_time maximum_connection_timeout( 32 * 1000 * 1000 );
999 static constexpr delta_time minimum_connection_timeout( 100 * 1000 );
1000 static constexpr auto max_slave_latency = 499;

1003 && transmit_window_size_ <= connection_interval_
1004 && connection_timeout_ >= minimum_connection_timeout

1007 && slave_latency_ <= max_slave_latency;

1056 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
1057 typename link_layer< Server, ScheduledRadio, Options... >::ll_result link_layer< Server, ScheduledRadio, Options... >::handle_received_data()
1058 {
1059 ll_result result = handle_pending_ll_control();
1060
1061 if ( result != ll_result::go_ahead || !defered_ll_control_pdu_.empty() )
1062 return result;
1063
1064 for ( auto pdu = this->next_ll_l2cap_received(); pdu.size != 0; )
1065 {
1066 const auto llid = layout_t::header( pdu ) & 0x03;
1067
1068 read_buffer output = { nullptr, 0 };
1069
1070 if ( llid == ll_control_pdu_code )
1071 {
1072 output = this->allocate_ll_transmit_buffer( maximum_ll_payload_size );
1073 if ( output.size )
1074 result = handle_ll_control_data( pdu, output );
1075 }
1076 else if ( llid == lld_data_pdu_code && state_ != state::disconnecting )
1077 {
1078 // TODO: Bug #67 will be too small for l2cap
1079 output = this->allocate_l2cap_transmit_buffer( connection_details_.negotiated_mtu() );
1080 if ( output.size )
1081 result = handle_l2cap( pdu, output );
1082 }
1083
1084 if ( output.size )
1085 {
1086 this->free_ll_l2cap_received();
1087 pdu = this->next_ll_l2cap_received();
1088 }
1089 else
1090 {
1091 pdu.size = 0;
1092 }
1093 }
1094
1095 return result;
1096 }

1122 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
1123 typename link_layer< Server, ScheduledRadio, Options... >::ll_result link_layer< Server, ScheduledRadio, Options... >::handle_ll_control_data( const write_buffer& pdu, read_buffer write )
1124 {
1125 ll_result result = ll_result::go_ahead;
1126 bool commit = true;
1127
1128 assert( write.size >= radio_t::min_buffer_size );
1129
1130 const std::uint8_t* const body = layout_t::body( pdu ).first;
1131 const std::uint16_t header = layout_t::header( pdu );
1132 std::uint8_t* const write_body = layout_t::body( write ).first;
1133
1134 if ( ( header & 0x3 ) == ll_control_pdu_code )
1135 {
1136 const std::uint8_t size = header >> 8;
1137 const std::uint8_t opcode = size > 0 ? *body : 0xff;
1138
1139 if ( opcode == LL_CONNECTION_UPDATE_REQ && size == 12 )
1140 {
1141 defered_conn_event_counter_ = read_16( &body[ 10 ] );
1142 commit = false;
1143
1144 if ( static_cast< std::uint16_t >( defered_conn_event_counter_ - conn_event_counter_ ) & 0x8000
1145 || defered_conn_event_counter_ == conn_event_counter_ )
1146 {
1147 result = ll_result::disconnect;
1148 }
1149 else
1150 {
1151 defered_ll_control_pdu_ = pdu;
1152 }
1153 }
1154 else if ( opcode == LL_TERMINATE_IND && size == 2 )
1155 {
1156 commit = false;
1157 result = ll_result::disconnect;
1158 }
1159 else if ( opcode == LL_VERSION_IND && size == 6 )
1160 {
1161 if ( body[ 1 ] <= LL_VERSION_40 )
1162 used_features_ = used_features_ & ~link_layer_feature::connection_parameters_request_procedure;
1163
1164 fill< layout_t >( write, {
1165 ll_control_pdu_code, 6, LL_VERSION_IND,
1166 LL_VERSION_NR, 0x69, 0x02, 0x00, 0x00
1167 } );
1168 }
1169 else if ( opcode == LL_CHANNEL_MAP_REQ && size == 8 )
1170 {
1171 defered_conn_event_counter_ = read_16( &body[ 6 ] );
1172 commit = false;
1173
1174 if ( static_cast< std::uint16_t >( defered_conn_event_counter_ - conn_event_counter_ ) & 0x8000 )
1175 {
1176 result = ll_result::disconnect;
1177 }
1178 else
1179 {
1180 defered_ll_control_pdu_ = pdu;
1181 }
1182 }
1183 else if ( opcode == LL_PING_REQ && size == 1 )
1184 {
1185 fill< layout_t >( write, { ll_control_pdu_code, 1, LL_PING_RSP } );
1186 }
1187 else if ( opcode == LL_FEATURE_REQ && size == 9 )
1188 {
1189 used_features_ = used_features_ & body[ 1 ];
1190
1191 fill< layout_t >( write, {
1192 ll_control_pdu_code, 9,
1193 LL_FEATURE_RSP,
1194 used_features_,
1195 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00
1196 } );
1197 }
1198 else if ( ( opcode == LL_UNKNOWN_RSP && size == 2 && body[ 1 ] == LL_CONNECTION_PARAM_REQ )
1199 || opcode == LL_REJECT_IND
1200 || ( opcode == LL_REJECT_IND_EXT && body[ 1 ] == LL_CONNECTION_PARAM_REQ ) )
1201 {
1202 if ( connection_parameters_request_running_ )
1203 {
1204 connection_parameters_request_running_ = false;
1205
1206 if ( signaling_channel_t::connection_parameter_update_request(
1207 proposed_interval_min_,
1208 proposed_interval_max_,
1209 proposed_latency_,
1210 proposed_timeout_ ) )
1211 {
1212 this->wake_up();
1213 }
1214 }
1215
1216 if ( opcode == LL_UNKNOWN_RSP )
1217 used_features_ = used_features_ & ~link_layer_feature::connection_parameters_request_procedure;
1218
1219 commit = false;
1220 }
1221 else if ( opcode == LL_CONNECTION_PARAM_REQ && size == 24 )
1222 {
1223 fill< layout_t >( write, { ll_control_pdu_code, size, LL_CONNECTION_PARAM_RSP } );
1224
1225 std::copy( &body[ 1 ], &body[ 1 + size - 1 ], &write_body[ 1 ] );
1226 }
1227 else if ( this->handle_encryption_pdus( opcode, size, pdu, write ) )
1228 {
1229 // all encryption PDU handled in handle_encryption_pdus()
1230 }
1231 else if ( opcode != LL_UNKNOWN_RSP )
1232 {
1233 fill< layout_t >( write, { ll_control_pdu_code, 2, LL_UNKNOWN_RSP, opcode } );
1234 }
1235 else
1236 {
1237 commit = false;
1238 }
1239
1240 if ( commit )
1241 this->commit_ll_transmit_buffer( write );
1242 }
1243
1244 return result;
1245 }

1247 template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >

1256 std::uint8_t* out_body = layout_t::body( output ).first;
1257

1265

1269 else if ( l2cap_channel == l2cap_signaling_channel )
