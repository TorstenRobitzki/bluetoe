129 void signaling_channel< Options... >::signaling_channel_input( const std::uint8_t* input, std::size_t in_size, std::uint8_t* output, std::size_t& out_size )
130 {
131 const std::uint8_t code = in_size > 0 ? input[ 0 ] : 0;
132
133 if ( code == connection_parameter_update_response_code && pending_status_ == transmitted )
134 {
135 pending_status_ = idle;
136 identifier_ = static_cast< std::uint8_t >( identifier_ + 1 );
137
138 if ( identifier_ == invalid_identifier )
139 identifier_ = static_cast< std::uint8_t >( identifier_ + 1 );
140
141 out_size = 0;
142 }
143 else
144 {

146 }

151 {
152 static constexpr std::size_t pdu_size = 8 + 4;
153 assert( out_size >= pdu_size );
154
155 if ( pending_status_ == queued )
156 {
157 pending_status_ = transmitted;
158
159 out_size = pdu_size;
160 output[ 0 ] = connection_parameter_update_request_code;
161 output[ 1 ] = identifier_;
162 output[ 2 ] = pdu_size - 4;
163 output[ 3 ] = 0;
164 output[ 4 ] = static_cast< std::uint8_t >( interval_min_ );
165 output[ 5 ] = static_cast< std::uint8_t >( interval_min_ >> 8 );

204 {
205 static constexpr std::size_t pdu_size = 6;
206 static constexpr std::uint16_t command_not_understood = 0;
207 assert( out_size >= pdu_size );
208 out_size = 0;
209
210 if ( in_size < 2 )
211 return;
212
213 const std::uint8_t identifier = input[ 1 ];
214
215 if ( identifier == invalid_identifier )
216 return;
217
218 out_size = pdu_size;
219 output[ 0 ] = command_reject_code;
220 output[ 1 ] = identifier;
221 output[ 2 ] = 2; // length
222 output[ 3 ] = 0; // length
223 output[ 4 ] = static_cast< std::uint8_t >( command_not_understood );
224 output[ 5 ] = static_cast< std::uint8_t >( command_not_understood >> 8 );
