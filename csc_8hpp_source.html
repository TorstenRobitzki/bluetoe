311  std::pair< std::uint8_t, bool > csc_write_control_point( std::size_t write_size, const std::uint8_t* value, Handler& handler )
312  {
313  if ( write_size < 1 )
314  return std::make_pair( error_codes::invalid_handle, false );
315 
316  if ( procedure_in_progress_ )
317  return std::make_pair( error_codes::procedure_already_in_progress, false );
318 
319  procedure_in_progress_ = true;
320  current_opcode_ = *value;
321  ++value;
322 
323  switch ( current_opcode_ )
324  {
325  case set_cumulative_value_opcode:
326  {
327  if ( write_size != 1 + 4 )
328  return std::make_pair( error_codes::invalid_pdu, false );
329 
330  handler.set_cumulative_wheel_revolutions( bluetoe::details::read_32bit( value ) );
331  return std::make_pair( error_codes::success, false );
332  }
333  case request_supported_sensor_locations_opcode:
334  {
335  if ( write_size != 1 )
336  return std::make_pair( error_codes::invalid_pdu, false );
337 
338  return std::make_pair( error_codes::success, true );
339  }
340  case update_sensor_location_opcode:
341  {
342  if ( write_size != 1 + 1 )
343  return std::make_pair( error_codes::invalid_pdu, false );
344 
345  this->set_sensor_position( *value);
346 
347  return std::make_pair( error_codes::success, true );
348  }
349  default:
350  // according to the spec with have to response with success and then indicate an error
351  return std::make_pair( error_codes::success, true );
352  }
