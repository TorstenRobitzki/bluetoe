189  {
190  bool ignore_first = true;
191 
192  // loop over all entries in a circle
193  for ( std::size_t i = next_; ignore_first || i != next_; i = ( i + 1 ) % Size )
194  {
195  ignore_first = false;
196  auto entry = at( i );
197 
198  if ( entry & indication_bit && outstanding_confirmation == no_outstanding_indicaton )
199  {
200  outstanding_confirmation = i + offset;
201  next_ = ( i + 1 ) % Size;
202  remove( i, indication_bit );
203  return { indication, i + offset };
204  }
205  else if ( entry & notification_bit )
206  {
207  next_ = ( i + 1 ) % Size;
208  remove( i, notification_bit );
209  return { notification, i + offset };
