385  template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
386  void link_layer< Server, ScheduledRadio, Options... >::adv_received( const read_buffer& receive )
387  {
388  assert( state_ == state::advertising );
389 
390  device_address remote_address;
391  const bool connection_request_received = this->handle_adv_receive( receive, remote_address );
392 
393  if ( connection_request_received
394  && channels_.reset( &receive.buffer[ 30 ], receive.buffer[ 35 ] & 0x1f )
395  && parse_timing_parameters_from_connect_request( receive ) )
396  {
397  state_ = state::connecting;
398  current_channel_index_ = 0;
399  conn_event_counter_ = 0;
400  cumulated_sleep_clock_accuracy_ = sleep_clock_accuracy( receive ) + device_sleep_clock_accuracy::accuracy_ppm;
401  timeouts_til_connection_lost_ = num_windows_til_timeout - 1;
402  used_features_ = supported_features;
403  connection_parameters_request_pending_ = false;
404  connection_parameters_request_running_ = false;
405 
406  this->set_access_address_and_crc_init(
407  read_32( &receive.buffer[ 14 ] ),
408  read_24( &receive.buffer[ 18 ] ) );
409 
410  const delta_time window_start = transmit_window_offset_ - transmit_window_offset_.ppm( cumulated_sleep_clock_accuracy_ );
411  delta_time window_end = transmit_window_offset_ + transmit_window_size_;
412 
413  window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
414 
415  this->reset();
416  this->schedule_connection_event(
417  channels_.data_channel( current_channel_index_ ),
418  window_start,
419  window_end,
420  connection_interval_ );
421 
422  this->connection_request( connection_addresses( address_, remote_address ) );
423  this->handle_stop_advertising();
424 
425  connection_details_ = notification_queue_t( details::mtu_size< Options... >::mtu );
426  }

527  template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
528  void link_layer< Server, ScheduledRadio, Options... >::wait_for_connection_event()
529  {
530  delta_time window_start;
531  delta_time window_end;
532 
533  if ( state_ == state::connecting )
534  {
535  const delta_time window_target = connection_interval_ * ( num_windows_til_timeout - timeouts_til_connection_lost_ - 1 );
536 
537  window_start = transmit_window_offset_ + window_target;
538  window_end = window_start + transmit_window_size_;
539 
540  window_start -= window_start.ppm( cumulated_sleep_clock_accuracy_ );
541  window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
542  }
543  else if ( state_ == state::connection_update )
544  {
545  window_start = connection_interval_old_ + transmit_window_offset_;
546  window_end = window_start + transmit_window_size_;
547 
548  window_start -= window_start.ppm( cumulated_sleep_clock_accuracy_ );
549  window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
550  }
551  else
552  {
553  const delta_time window_target = connection_interval_ * ( max_timeouts_til_connection_lost_ - timeouts_til_connection_lost_ + 1 );
554  const delta_time window_size = window_target.ppm( cumulated_sleep_clock_accuracy_ );
555 
556  window_start = window_target - window_size;
557  window_end = window_target + window_size;
558  }
559 
560  const delta_time time_till_next_event = this->schedule_connection_event(
561  channels_.data_channel( current_channel_index_ ),
562  window_start,
563  window_end,
564  connection_interval_ );
565 
566  connection_event_callback::call_connection_event_callback( time_till_next_event );
567  }

569  template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
570  void link_layer< Server, ScheduledRadio, Options... >::transmit_notifications()
571  {
572  // first check if we have memory to transmit the message, or otherwise notifications would get lost
573  auto out_buffer = this->allocate_transmit_buffer();
574 
575  if ( out_buffer.empty() )
576  return;
577 
578  const auto notification = connection_details_.dequeue_indication_or_confirmation();
579 
580  if ( notification.first != notification_queue_t::entry_type::empty )
581  {
582  std::size_t out_size = out_buffer.size - all_header_size;
583 
584  if ( notification.first == notification_queue_t::entry_type::notification )
585  {
586  server_->notification_output(
587  &out_buffer.buffer[ all_header_size ],
588  out_size,
589  connection_details_,
590  notification.second
591  );
592  }
593  else
594  {
595  server_->indication_output(
596  &out_buffer.buffer[ all_header_size ],
597  out_size,
598  connection_details_,
599  notification.second
600  );
601 
602  // if no output is generate, confirm the indication, or we will wait for ever
603  if ( out_size == 0 )
604  connection_details_.indication_confirmed();
605 
606  }
607 
608  if ( out_size )
609  {
610  out_buffer.buffer[ 0 ] = lld_data_pdu_code;
611  out_buffer.buffer[ 1 ] = static_cast< std::uint8_t >( out_size + l2cap_header_size );
612  out_buffer.buffer[ 2 ] = static_cast< std::uint8_t >( out_size );
613  out_buffer.buffer[ 3 ] = 0;
614  out_buffer.buffer[ 4 ] = static_cast< std::uint8_t >( l2cap_att_channel );
615  out_buffer.buffer[ 5 ] = static_cast< std::uint8_t >( l2cap_att_channel >> 8 );
616 
617  this->commit_transmit_buffer( out_buffer );
618  }
619  }
620  }

910  }
911  else if ( opcode == LL_FEATURE_REQ && size == 9 )
912  {

914

925  {
926  connection_parameters_request_running_ = false;

935  }
936  }

940  }

947  else if ( opcode != LL_UNKNOWN_RSP )

950  }
951  else
952  {

954  }
955 
956  if ( commit )
957  this->commit_transmit_buffer( write );
958  }
959 
960  return result;
