121  bool handle_encryption_pdus( std::uint8_t opcode, std::uint8_t size, const write_buffer& pdu, read_buffer write )
122  {
123  using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
124 
125  if ( opcode == LinkLayer::LL_ENC_REQ && size == 23 )
126  {
127  encryption_in_progress_ = true;
128  fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1 + 8 + 4, LinkLayer::LL_ENC_RSP } );
129 
130  const std::uint8_t* const pdu_body = layout_t::body( pdu ).first;
131 
132  const std::uint64_t rand = LinkLayer::read_64( pdu_body +1 );
133  const std::uint16_t ediv = LinkLayer::read_16( pdu_body +9 );
134  const std::uint64_t skdm = LinkLayer::read_64( pdu_body +11 );
135  const std::uint32_t ivm = LinkLayer::read_32( pdu_body +19 );
136  std::uint64_t skds = 0;
137  std::uint32_t ivs = 0;
138 
139  bluetoe::details::uint128_t key;
140  std::tie( has_key_, key ) = that().connection_details_.find_key( ediv, rand );
141 
142  // setup encryption
143  std::tie( skds, ivs ) = that().setup_encryption( key, skdm, ivm );
144 
145  std::uint8_t* write_body = layout_t::body( write ).first;
146  bluetoe::details::write_64bit( &write_body[ 1 ], skds );
147  bluetoe::details::write_32bit( &write_body[ 9 ], ivs );
148  }
149  else if ( opcode == LinkLayer::LL_START_ENC_RSP && size == 1 )
150  {
151  fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_RSP } );
152  that().start_transmit_encrypted();
153  that().connection_details_.is_encrypted( true );
154  }
155  else if ( opcode == LinkLayer::LL_PAUSE_ENC_REQ && size == 1 )
156  {
157  fill< layout_t >( write, { LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_PAUSE_ENC_RSP } );
158  that().stop_receive_encrypted();
159  that().connection_details_.is_encrypted( false );
160  }
161  else if ( opcode == LinkLayer::LL_PAUSE_ENC_RSP && size == 1 )
162  {
163  that().stop_transmit_encrypted();
164  that().connection_details_.is_encrypted( false );
165 
166  return false;
167  }
168  else
169  {
170  return false;
171  }
172 
173  return true;
174  }
175 
176  void transmit_pending_security_pdus()
177  {
178  using layout_t = typename pdu_layout_by_radio< typename LinkLayer::radio_t >::pdu_layout;
179 
180  if ( !encryption_in_progress_ )
181  return;
182 
183  auto out_buffer = that().allocate_transmit_buffer();
184  if ( out_buffer.empty() )
185  return;
186 
187  if ( has_key_ )
188  {
189  fill< layout_t >( out_buffer, {
190  LinkLayer::ll_control_pdu_code, 1, LinkLayer::LL_START_ENC_REQ } );
191 
192  that().start_receive_encrypted();
193  that().commit_transmit_buffer( out_buffer );
194  }
195  else
196  {
197  fill< layout_t >( out_buffer, {
198  LinkLayer::ll_control_pdu_code, 2, LinkLayer::LL_REJECT_IND, LinkLayer::err_pin_or_key_missing } );
199 
200  that().commit_transmit_buffer( out_buffer );
201  }
202 
203  encryption_in_progress_ = false;
204  }
205 
206  void reset_encryption()
207  {
208  that().connection_details_.is_encrypted( false );
209  that().stop_receive_encrypted();
210  that().stop_transmit_encrypted();
211  }
212 
213  private:
214  bool has_key_;
215  bool encryption_in_progress_;
216  };

575  template < class Server, template < std::size_t, std::size_t, class > class ScheduledRadio, typename ... Options >
576  void link_layer< Server, ScheduledRadio, Options... >::adv_received( const read_buffer& receive )
577  {
578  assert( state_ == state::advertising );
579 
580  device_address remote_address;
581  const bool connection_request_received = this->handle_adv_receive( receive, remote_address );
582 
583  if ( connection_request_received )
584  {
585  const std::uint8_t* const body = layout_t::body( receive ).first;
586 
587  if ( channels_.reset( &body[ 28 ], body[ 33 ] & 0x1f )
588  && parse_timing_parameters_from_connect_request( body ) )
589  {
590  state_ = state::connecting;
591  current_channel_index_ = 0;
592  conn_event_counter_ = 0;
593  cumulated_sleep_clock_accuracy_ = sleep_clock_accuracy( body ) + device_sleep_clock_accuracy::accuracy_ppm;
594  timeouts_til_connection_lost_ = num_windows_til_timeout - 1;
595  used_features_ = supported_features;
596  connection_parameters_request_pending_ = false;
597  connection_parameters_request_running_ = false;
598 
599  this->set_access_address_and_crc_init( read_32( &body[ 12 ] ), read_24( &body[ 16 ] ) );
600 
601  const delta_time window_start = transmit_window_offset_ - transmit_window_offset_.ppm( cumulated_sleep_clock_accuracy_ );
602  delta_time window_end = transmit_window_offset_ + transmit_window_size_;
603 
604  window_end += window_end.ppm( cumulated_sleep_clock_accuracy_ );
605 
606  this->reset();
607  this->schedule_connection_event(
608  channels_.data_channel( current_channel_index_ ),
609  window_start,
610  window_end,
611  connection_interval_ );
612 
613  this->connection_request( connection_addresses( address_, remote_address ) );
614  this->handle_stop_advertising();
615 
616  connection_details_ = connection_details_t( std::size_t{ details::mtu_size< Options... >::mtu } );
617  connection_details_.remote_connection_created( remote_address );
618  }
619  }

644  force_disconnect();

646  else
647  {
648  wait_for_connection_event();
649  }

666 
667  if ( state_ != state::disconnecting )
668  {
669  state_ = state::connected;

674  ++conn_event_counter_;
675 
676  if ( handle_received_data() == ll_result::disconnect || send_control_pdus() == ll_result::disconnect )
677  {
678  force_disconnect();
679  }
680  else
681  {
682  this->transmit_pending_security_pdus();
683  wait_for_connection_event();
684  }

1051 
1052  assert( write.size >= radio_t::min_buffer_size );
1053

1055  const std::uint16_t header = layout_t::header( pdu );
1056  std::uint8_t* const write_body = layout_t::body( write ).first;
1057 
1058  if ( ( header & 0x3 ) == ll_control_pdu_code )
1059  {
1060 
1061  const std::uint8_t size = header >> 8;
1062  const std::uint8_t opcode = size > 0 ? *body : 0xff;
1063 
1064  if ( opcode == LL_CONNECTION_UPDATE_REQ && size == 12 )
1065  {
1066  defered_conn_event_counter_ = read_16( &body[ 10 ] );
1067  commit = false;
1068 
1069  if ( static_cast< std::uint16_t >( defered_conn_event_counter_ - conn_event_counter_ ) & 0x8000
1070  || defered_conn_event_counter_ == conn_event_counter_ )
1071  {
1072  result = ll_result::disconnect;
1073  }
1074  else
1075  {
1076  defered_ll_control_pdu_ = pdu;
1077  }
1078  }
1079  else if ( opcode == LL_TERMINATE_IND && size == 2 )
1080  {
1081  commit = false;
1082  result = ll_result::disconnect;
1083  }
1084  else if ( opcode == LL_VERSION_IND && size == 6 )

1086  if ( body[ 1 ] <= LL_VERSION_40 )
1087  used_features_ = used_features_ & ~link_layer_feature::connection_parameters_request_procedure;

1098

1101  result = ll_result::disconnect;
1102  }

1104  {
1105  defered_ll_control_pdu_ = pdu;
1106  }
1107  }
1108  else if ( opcode == LL_PING_REQ && size == 1 )
1109  {
1110  fill< layout_t >( write, { ll_control_pdu_code, 1, LL_PING_RSP } );
1111  }
1112  else if ( opcode == LL_FEATURE_REQ && size == 9 )
1113  {
1114  used_features_ = used_features_ & body[ 1 ];
1115 
1116  fill< layout_t >( write, {
1117  ll_control_pdu_code, 9,
1118  LL_FEATURE_RSP,
1119  used_features_,
1120  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00
1121  } );
1122  }
1123  else if ( opcode == LL_UNKNOWN_RSP && size == 2 && body[ 1 ] == LL_CONNECTION_PARAM_REQ )
1124  {
1125  if ( connection_parameters_request_running_ )
1126  {
1127  connection_parameters_request_running_ = false;
1128 
1129  if ( signaling_channel_t::connection_parameter_update_request(
1130  proposed_interval_min_,
1131  proposed_interval_max_,
1132  proposed_latency_,
1133  proposed_timeout_ ) )
1134  {
1135  this->wake_up();
1136  }
1137  }
1138 
1139  used_features_ = used_features_ & ~link_layer_feature::connection_parameters_request_procedure;
1140  commit = false;
1141  }
1142  else if ( opcode == LL_CONNECTION_PARAM_REQ && size == 24 )
1143  {
1144  fill< layout_t >( write, { ll_control_pdu_code, size, LL_CONNECTION_PARAM_RSP } );
1145 
1146  std::copy( &body[ 1 ], &body[ 1 + size - 1 ], &write_body[ 1 ] );
1147  }
1148  else if ( this->handle_encryption_pdus( opcode, size, pdu, write ) )
1149  {
1150  // all encryption PDU handled in handle_encryption_pdus()
1151  }
1152  else if ( opcode != LL_UNKNOWN_RSP )
1153  {
1154  fill< layout_t >( write, { ll_control_pdu_code, 2, LL_UNKNOWN_RSP, opcode } );
1155  }
1156  else
1157  {
1158  commit = false;
1159  }
1160 
1161  if ( commit )
1162  this->commit_transmit_buffer( write );
1163  }
1164 
1165  return result;
1166  }
